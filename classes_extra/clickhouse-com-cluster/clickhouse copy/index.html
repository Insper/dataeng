    ClickHouse

    O ClickHouse é um sistema de gerenciamento de banco de dados (SGBD) analítico columnar (OLAP) de código aberto, desenvolvido pela Yandex. É otimizado especificamente para consultas analíticas em tempo real sobre grandes volumes de dados.

    Info

    O ClickHouse é conhecido por sua excepcional velocidade em consultas analíticas.

    Utiliza armazenamento columnar e processamento paralelo massivo (MPP) para alta performance.

    Características principais

    • Columnar storage: Armazena dados por colunas, otimizando consultas analíticas
    • Compressão avançada: Algoritmos de compressão específicos para cada tipo de dado
    • SQL completo: Suporte robusto ao SQL com extensões para análise avançada
    • Distribuído: Escalabilidade horizontal nativa com clusters
    • Real-time: Inserções e consultas em tempo real
    • Formatos diversos: Suporte nativo a Parquet, CSV, JSON e outros formatos

    Info!

    O ClickHouse só permite acesso a arquivos dentro do diretório user_files por questões de segurança.

    Ao fazer ingestão, acessaremos ./dados via /var/lib/clickhouse/user_files/ dentro do container.

    Atenção!

    Observe que, com a definição do volume clickhouse-data, teremos persistência dos dados mesmo após a parada ou remoção do container.

    Confira os volumes com (só irá aparecer após você iniciar o container):

    $ docker volume ls

    Consultas analíticas avançadas

    Window Functions

    O ClickHouse suporta window functions para análises sofisticadas.

    Exercise

    Utilize window functions para análises avançadas:

    -- Ranking das estações mais populares por mês
SELECT 
    toYYYYMM(start_date) as month,
    start_station_name,
    COUNT(*) as trip_count,
    ROW_NUMBER() OVER (PARTITION BY toYYYYMM(start_date) ORDER BY COUNT(*) DESC) as rank
FROM trip 
GROUP BY month, start_station_name 
ORDER BY month, rank
LIMIT 20;

    Exercise

    Calcule médias móveis:

    -- Média móvel de viagens por dia (janela de 7 dias)
SELECT 
    date,
    daily_trips,
    AVG(daily_trips) OVER (ORDER BY date ROWS 6 PRECEDING) as moving_avg_7days
FROM (
    SELECT 
        toDate(start_date) as date,
        COUNT(*) as daily_trips
    FROM trip 
    GROUP BY date
    ORDER BY date
)
ORDER BY date;

    Importando dados Parquet

    O ClickHouse tem suporte nativo ao formato Parquet.

    Exercise

    Crie uma tabela lendo diretamente do Parquet (com conversão de tipos):

    CREATE TABLE trip_parquet
(
    id String,
    duration Int32,
    start_date DateTime,
    start_station_name String,
    start_station_id Int32,
    end_date DateTime,
    end_station_name String,
    end_station_id Int32,
    bike_id String,
    subscription_type String,
    zip_code String
)
ENGINE = MergeTree
PARTITION BY toYYYYMM(start_date)
ORDER BY (start_date, start_station_id);

-- Then insert the transformed data
INSERT INTO trip_parquet
SELECT
    id,
    duration,
    parseDateTimeBestEffort(start_date) AS start_date,
    start_station_name,
    start_station_id,
    parseDateTimeBestEffort(end_date) AS end_date,
    end_station_name,
    end_station_id,
    bike_id,
    subscription_type,
    zip_code
FROM file('parquets/trip.parquet', 'Parquet');

    Conversão de tipos

    Como o ClickHouse pode interpretar colunas de data como String ao ler Parquet, usamos parseDateTimeBestEffort() para converter adequadamente os tipos de data tanto na partição quanto na ordenação.

    O ClickHouse possui funções geoespaciais avançadas.

    Exercise

    Faça análises baseadas em localização:

    -- Junção para obter coordenadas das estações
CREATE TABLE trip_with_coords
ENGINE = MergeTree
ORDER BY (start_station_id, end_station_id)
AS SELECT 
    t.*,
    s1.lat as start_lat,
    s1.long as start_long,
    s2.lat as end_lat, 
    s2.long as end_long
FROM trip t
LEFT JOIN station s1 ON t.start_station_id = s1.id
LEFT JOIN station s2 ON t.end_station_id = s2.id;

    Exercise

    Calcule distâncias entre estações:

    -- Distância aproximada entre estações (em metros)
SELECT 
    start_station_name,
    end_station_name,
    COUNT(*) as trip_count,
    geoDistance(start_long, start_lat, end_long, end_lat) as distance_meters
FROM trip_with_coords 
WHERE start_lat IS NOT NULL AND end_lat IS NOT NULL
GROUP BY start_station_name, end_station_name, start_lat, start_long, end_lat, end_long
HAVING trip_count > 10
ORDER BY distance_meters DESC
LIMIT 10;

    Criando o schema ClickHouse

    Exercise

    Crie um novo arquivo 02-etl/sql/0001-ddl-clickhouse.sql com o schema otimizado para ClickHouse:

    -- Database já é criado pelo container
-- USE vendas_dw;

-- Tabela de cidades otimizada para ClickHouse
CREATE TABLE IF NOT EXISTS cidades
(
    id UInt32,
    nome LowCardinality(String),
    uf LowCardinality(String),
    created_at DateTime DEFAULT now(),
    updated_at DateTime DEFAULT now()
)
ENGINE = MergeTree()
ORDER BY id;

-- Tabela de clientes otimizada
CREATE TABLE IF NOT EXISTS clientes
(
    id UInt32,
    nome String,
    email String,
    telefone String,
    cidade_id UInt32,
    created_at DateTime DEFAULT now(),
    updated_at DateTime DEFAULT now()
)
ENGINE = MergeTree()
PARTITION BY toYYYYMM(created_at)
ORDER BY (created_at, id);

-- Tabela de produtos otimizada
CREATE TABLE IF NOT EXISTS produtos
(
    id UInt32,
    nome LowCardinality(String),
    categoria LowCardinality(String),
    preco Decimal(10, 2),
    created_at DateTime DEFAULT now(),
    updated_at DateTime DEFAULT now()
)
ENGINE = MergeTree()
ORDER BY (categoria, id);

-- Tabela de vendas particionada por data
CREATE TABLE IF NOT EXISTS vendas
(
    id UInt64,
    cliente_id UInt32,
    data_venda DateTime,
    valor_total Decimal(10, 2),
    status LowCardinality(String),
    created_at DateTime DEFAULT now(),
    updated_at DateTime DEFAULT now()
)
ENGINE = MergeTree()
PARTITION BY toYYYYMM(data_venda)
ORDER BY (data_venda, cliente_id);

-- Tabela de itens de venda
CREATE TABLE IF NOT EXISTS itens_venda
(
    id UInt64,
    venda_id UInt64,
    produto_id UInt32,
    quantidade UInt32,
    preco_unitario Decimal(10, 2),
    subtotal Decimal(10, 2),
    created_at DateTime DEFAULT now(),
    updated_at DateTime DEFAULT now()
)
ENGINE = MergeTree()
PARTITION BY toYYYYMM(created_at)
ORDER BY (venda_id, produto_id);

    Exercise

    Quais otimizações foram aplicadas no schema do ClickHouse em comparação com o PostgreSQL?

    Answer

    As principais otimizações aplicadas foram:

    1. LowCardinality: Para campos com poucos valores únicos (nome, categoria, uf, status)
    2. Particionamento: Tabelas particionadas por mês usando toYYYYMM()
    3. Tipos específicos: UInt32/UInt64 para IDs, Decimal para valores monetários
    4. Ordenação otimizada: ORDER BY considera padrões de consulta (data, categoria, etc.)
    5. Remoção de constraints: Sem chaves estrangeiras ou checks para melhor performance de inserção

    Adaptando o script de inicialização

    Exercise

    Crie um novo arquivo 02-etl/src/init_clickhouse.py:

    import clickhouse_connect
import time
import os
from pathlib import Path

def wait_for_clickhouse(max_retries=30):
    """Aguarda o ClickHouse ficar disponível"""
    for attempt in range(max_retries):
        try:
            client = clickhouse_connect.get_client(
                host=os.getenv('CLICKHOUSE_HOST', 'clickhouse-dw'),
                port=8123,
                username=os.getenv('CLICKHOUSE_USER', 'dw_user'),
                password=os.getenv('CLICKHOUSE_PASSWORD', 'dw_pass'),
                database=os.getenv('CLICKHOUSE_DB', 'vendas_dw')
            )

            # Teste simples de conexão
            result = client.query('SELECT 1')
            print("✅ ClickHouse está disponível!")
            return client

        except Exception as e:
            print(f"⏳ Tentativa {attempt + 1}/{max_retries} - Aguardando ClickHouse... ({e})")
            time.sleep(2)

    raise Exception("❌ ClickHouse não ficou disponível")

def init_clickhouse_schema():
    """Inicializa o schema no ClickHouse"""
    client = wait_for_clickhouse()

    # Lê o arquivo DDL
    ddl_file = Path(__file__).parent.parent / 'sql' / '0001-ddl-clickhouse.sql'

    with open(ddl_file, 'r', encoding='utf-8') as f:
        ddl_content = f.read()

    # Executa os comandos DDL
    commands = [cmd.strip() for cmd in ddl_content.split(';') if cmd.strip()]

    for command in commands:
        if command and not command.startswith('--'):
            try:
                client.command(command)
                print(f"✅ Executado: {command[:50]}...")
            except Exception as e:
                print(f"❌ Erro ao executar comando: {e}")
                print(f"   Comando: {command}")

    print("🎉 Schema do ClickHouse inicializado com sucesso!")

if __name__ == "__main__":
    init_clickhouse_schema()

    import psycopg2
import clickhouse_connect
import pandas as pd
import time
import os
from datetime import datetime, timedelta

class ETLVendasClickHouse:
    def __init__(self):
        # Conexão com origem (PostgreSQL)
        self.pg_conn = psycopg2.connect(
            host=os.getenv('POSTGRES_HOST', 'postgres-app'),
            database=os.getenv('POSTGRES_DB', 'vendas'),
            user=os.getenv('POSTGRES_USER', 'vendas_user'),
            password=os.getenv('POSTGRES_PASSWORD', 'vendas_pass')
        )

        # Conexão com destino (ClickHouse)
        self.ch_client = clickhouse_connect.get_client(
            host=os.getenv('CLICKHOUSE_HOST', 'clickhouse-dw'),
            port=8123,
            username=os.getenv('CLICKHOUSE_USER', 'dw_user'),
            password=os.getenv('CLICKHOUSE_PASSWORD', 'dw_pass'),
            database=os.getenv('CLICKHOUSE_DB', 'vendas_dw')
        )

        self.last_sync = {}

    def extract_incremental(self, table_name, timestamp_column='created_at'):
        """Extrai dados incrementais do PostgreSQL"""
        last_sync_time = self.last_sync.get(table_name, datetime.now() - timedelta(minutes=5))

        query = f"""
        SELECT * FROM {table_name} 
        WHERE {timestamp_column} >= %s 
        ORDER BY {timestamp_column}
        """

        df = pd.read_sql(query, self.pg_conn, params=[last_sync_time])

        if not df.empty:
            # Atualiza último timestamp de sincronização
            self.last_sync[table_name] = df[timestamp_column].max()
            print(f"📥 Extraídos {len(df)} registros de {table_name}")

        return df

    def load_to_clickhouse(self, df, table_name):
        """Carrega dados no ClickHouse"""
        if df.empty:
            return

        try:
            # Converte colunas datetime para o formato esperado
            datetime_columns = df.select_dtypes(include=['datetime64']).columns
            for col in datetime_columns:
                df[col] = pd.to_datetime(df[col])

            self.ch_client.insert_df(table_name, df)
            print(f"📤 Carregados {len(df)} registros em {table_name}")

        except Exception as e:
            print(f"❌ Erro ao carregar {table_name}: {e}")

    def sync_table(self, table_name, timestamp_column='created_at'):
        """Sincroniza uma tabela completa"""
        print(f"🔄 Sincronizando tabela: {table_name}")

        # Extract
        df = self.extract_incremental(table_name, timestamp_column)

        # Load
        if not df.empty:
            self.load_to_clickhouse(df, table_name)
        else:
            print(f"ℹ️  Nenhum registro novo em {table_name}")

    def run_etl_cycle(self):
        """Executa um ciclo completo de ETL"""
        print(f"🚀 Iniciando ciclo ETL - {datetime.now()}")

        tables = [
            'cidades',
            'clientes', 
            'produtos',
            'vendas',
            'itens_venda'
        ]

        for table in tables:
            try:
                # Usa data_venda para vendas, created_at para outras
                timestamp_col = 'data_venda' if table == 'vendas' else 'created_at'
                self.sync_table(table, timestamp_col)
            except Exception as e:
                print(f"❌ Erro ao sincronizar {table}: {e}")

        print(f"✅ Ciclo ETL concluído - {datetime.now()}")
        print("-" * 50)

def main():
    etl = ETLVendasClickHouse()

    while True:
        try:
            etl.run_etl_cycle()
            # Aguarda 2 minutos até o próximo ciclo
            time.sleep(120)

        except KeyboardInterrupt:
            print("\n🛑 ETL interrompido pelo usuário")
            break
        except Exception as e:
            print(f"❌ Erro geral no ETL: {e}")
            time.sleep(30)  # Aguarda antes de tentar novamente

if __name__ == "__main__":
    main()

    Criando views materializadas para análises

    Exercise

    Crie views materializadas para consultas frequentes:

    -- View para vendas diárias resumidas
CREATE MATERIALIZED VIEW vendas_diarias_mv
ENGINE = SummingMergeTree()
PARTITION BY toYYYYMM(data_venda)
ORDER BY (data_venda, cliente_id)
AS SELECT
    toDate(data_venda) as data_venda,
    cliente_id,
    count() as total_vendas,
    sum(valor_total) as receita_total
FROM vendas
GROUP BY data_venda, cliente_id;

-- View para produtos mais vendidos
CREATE MATERIALIZED VIEW produtos_vendidos_mv
ENGINE = SummingMergeTree()
ORDER BY (produto_id, toYYYYMM(data_venda))
AS SELECT
    iv.produto_id,
    toYYYYMM(v.data_venda) as mes,
    sum(iv.quantidade) as quantidade_vendida,
    sum(iv.subtotal) as receita_produto
FROM itens_venda iv
JOIN vendas v ON iv.venda_id = v.id
GROUP BY iv.produto_id, mes;
